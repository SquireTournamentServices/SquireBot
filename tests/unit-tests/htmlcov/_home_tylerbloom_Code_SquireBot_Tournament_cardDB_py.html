127 def updateCards(self) -> bool: 

128 compressedCacheName = self.cacheName + ".zip" 

129 resp = requests.get(self.url, timeout=7.0, data="", verify=False) 

130 

131 # Save zip file 

132 tmpFile = tmpFile = tempfile.TemporaryFile(mode="wb+", suffix="cardDB.py", prefix=compressedCacheName) 

133 for chunk in resp.iter_content(chunk_size=512 * 1024): 

134 if chunk: # filter out keep-alive new chunks 

135 tmpFile.write(chunk) 

136 

137 # Go to the start of the file before unzipping

141 zip = zipfile.ZipFile(tmpFile, "r")

143 

144 # Close file
