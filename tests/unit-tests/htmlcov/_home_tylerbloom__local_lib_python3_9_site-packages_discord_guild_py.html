245 def _add_role(self, role): 

246 # roles get added to the bottom (position 1, pos 0 is @everyone) 

247 # so since self.roles has the @everyone role, we can't increment 

248 # its position because it's stuck at position 0. Luckily x += False 

249 # is equivalent to adding 0. So we cast the position to a bool and 

250 # increment it. 

251 for r in self._roles.values(): 

252 r.position += (not r.is_default()) 

253 

254 self._roles[role.id] = role

256 def _remove_role(self, role_id): 

257 # this raises KeyError if it fails.. 

258 role = self._roles.pop(role_id) 

259 

260 # since it didn't, we can change the positions now 

261 # basically the same as above except we only decrement 

262 # the position if we're above the role we deleted. 

263 for r in self._roles.values(): 

264 r.position -= r.position > role.position 

265 

266 return role

783 def get_member_named(self, name): 

784 """Returns the first member found that matches the name provided. 

785 

786 The name can have an optional discriminator argument, e.g. "Jake#0001" 

787 or "Jake" will both do the lookup. However the former will give a more 

788 precise result. Note that the discriminator must have all 4 digits 

789 for this to work. 

790 

791 If a nickname is passed, then it is looked up via the nickname. Note 

792 however, that a nickname + discriminator combo will not lookup the nickname 

793 but rather the username + discriminator combo due to nickname + discriminator 

794 not being unique. 

795 

796 If no member is found, ``None`` is returned. 

797 

798 Parameters 

799 ----------- 

800 name: :class:`str` 

801 The name of the member to lookup with an optional discriminator.

807 then ``None`` is returned.

810 result = None 

811 members = self.members 

812 if len(name) > 5 and name[-5] == '#': 

813 # The 5 length is checking to see if #0000 is in the string, 

814 # as a#0000 has a length of 6, the minimum for a potential 

815 # discriminator lookup. 

816 potential_discriminator = name[-4:] 

817 

818 # do the actual lookup and return if found 

819 # if it isn't found then we'll do a full name lookup below. 

820 result = utils.get(members, name=name[:-5], discriminator=potential_discriminator) 

821 if result is not None:

825 return m.nick == name or m.name == name
