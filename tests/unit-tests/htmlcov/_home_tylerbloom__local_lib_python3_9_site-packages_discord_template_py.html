82 """Represents a Discord template. 

83 

84 .. versionadded:: 1.4 

85 

86 Attributes

89 The template code. 

90 uses: :class:`int`

92 name: :class:`str` 

93 The name of the template. 

94 description: :class:`str` 

95 The description of the template. 

96 creator: :class:`User` 

97 The creator of the template. 

98 created_at: :class:`datetime.datetime` 

99 When the template was created. 

100 updated_at: :class:`datetime.datetime` 

101 When the template was last updated (referred to as "last synced" in the client). 

102 source_guild: :class:`Guild` 

103 The source guild. 

104 """

106 def __init__(self, *, state, data): 

107 self._state = state 

108 self._store(data)

110 def _store(self, data): 

111 self.code = data['code'] 

112 self.uses = data['usage_count'] 

113 self.name = data['name'] 

114 self.description = data['description'] 

115 creator_data = data.get('creator') 

116 self.creator = None if creator_data is None else self._state.store_user(creator_data) 

117 

118 self.created_at = parse_time(data.get('created_at')) 

119 self.updated_at = parse_time(data.get('updated_at')) 

120 

121 id = _get_as_snowflake(data, 'source_guild_id') 

122 

123 guild = self._state._get_guild(id) 

124 

125 if guild is None: 

126 source_serialised = data['serialized_source_guild'] 

127 source_serialised['id'] = id 

128 state = _PartialTemplateState(state=self._state) 

129 guild = Guild(data=source_serialised, state=state) 

130 

131 self.source_guild = guild

137 async def create_guild(self, name, region=None, icon=None): 

138 """|coro| 

139 

140 Creates a :class:`.Guild` using the template. 

141 

142 Bot accounts in more than 10 guilds are not allowed to create guilds. 

143 

144 Parameters 

145 ---------- 

146 name: :class:`str` 

147 The name of the guild. 

148 region: :class:`.VoiceRegion` 

149 The region for the voice communication server. 

150 Defaults to :attr:`.VoiceRegion.us_west`. 

151 icon: :class:`bytes` 

152 The :term:`py:bytes-like object` representing the icon. See :meth:`.ClientUser.edit` 

153 for more details on what is expected. 

154 

155 Raises 

156 ------ 

157 HTTPException 

158 Guild creation failed. 

159 InvalidArgument 

160 Invalid icon image format given. Must be PNG or JPG. 

161 

162 Returns 

163 ------- 

164 :class:`.Guild` 

165 The guild created. This is not the same guild that is 

166 added to cache. 

167 """ 

168 if icon is not None: 

169 icon = _bytes_to_base64_data(icon) 

170 

171 region = region or VoiceRegion.us_west 

172 region_value = region.value 

173 

174 data = await self._state.http.create_from_template(self.code, name, region_value, icon) 

175 return Guild(data=data, state=self._state)
