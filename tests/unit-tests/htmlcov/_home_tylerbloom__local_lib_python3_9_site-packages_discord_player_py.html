236class FFmpegOpusAudio(FFmpegAudio): 

237 """An audio source from FFmpeg (or AVConv). 

238 

239 This launches a sub-process to a specific input file given. However, rather than 

240 producing PCM packets like :class:`FFmpegPCMAudio` does that need to be encoded to 

241 Opus, this class produces Opus packets, skipping the encoding step done by the library. 

242 

243 Alternatively, instead of instantiating this class directly, you can use 

244 :meth:`FFmpegOpusAudio.from_probe` to probe for bitrate and codec information. This 

245 can be used to opportunistically skip pointless re-encoding of existing Opus audio data 

246 for a boost in performance at the cost of a short initial delay to gather the information. 

247 The same can be achieved by passing ``copy`` to the ``codec`` parameter, but only if you 

248 know that the input source is Opus encoded beforehand. 

249 

250 .. versionadded:: 1.3 

251 

252 .. warning:: 

253 

254 You must have the ffmpeg or avconv executable in your path environment 

255 variable in order for this to work. 

256 

257 Parameters 

258 ------------ 

259 source: Union[:class:`str`, :class:`io.BufferedIOBase`] 

260 The input that ffmpeg will take and convert to Opus bytes. 

261 If ``pipe`` is ``True`` then this is a file-like object that is 

262 passed to the stdin of ffmpeg. 

263 bitrate: :class:`int` 

264 The bitrate in kbps to encode the output to. Defaults to ``128``. 

265 codec: Optional[:class:`str`] 

266 The codec to use to encode the audio data. Normally this would be 

267 just ``libopus``, but is used by :meth:`FFmpegOpusAudio.from_probe` to 

268 opportunistically skip pointlessly re-encoding Opus audio data by passing 

269 ``copy`` as the codec value. Any values other than ``copy``, ``opus``, or 

270 ``libopus`` will be considered ``libopus``. Defaults to ``libopus``. 

271 

272 .. warning:: 

273 

274 Do not provide this parameter unless you are certain that the audio input is 

275 already Opus encoded. For typical use :meth:`FFmpegOpusAudio.from_probe` 

276 should be used to determine the proper value for this parameter. 

277 

278 executable: :class:`str` 

279 The executable name (and path) to use. Defaults to ``ffmpeg``. 

280 pipe: :class:`bool` 

281 If ``True``, denotes that ``source`` parameter will be passed 

282 to the stdin of ffmpeg. Defaults to ``False``. 

283 stderr: Optional[:term:`py:file object`] 

284 A file-like object to pass to the Popen constructor. 

285 Could also be an instance of ``subprocess.PIPE``. 

286 before_options: Optional[:class:`str`]

389 @classmethod 

390 async def probe(cls, source, *, method=None, executable=None): 

391 """|coro| 

392 

393 Probes the input source for bitrate and codec information. 

394 

395 Parameters 

396 ------------ 

397 source 

398 Identical to the ``source`` parameter for :class:`FFmpegOpusAudio`. 

399 method 

400 Identical to the ``method`` parameter for :meth:`FFmpegOpusAudio.from_probe`. 

401 executable: :class:`str` 

402 Identical to the ``executable`` parameter for :class:`FFmpegOpusAudio`. 

403 

404 Raises 

405 -------- 

406 AttributeError 

407 Invalid probe method, must be ``'native'`` or ``'fallback'``. 

408 TypeError 

409 Invalid value for ``probe`` parameter, must be :class:`str` or a callable. 

410 

411 Returns 

412 --------- 

413 Tuple[Optional[:class:`str`], Optional[:class:`int`]] 

414 A 2-tuple with the codec and bitrate of the input source. 

415 """ 

416 

417 method = method or 'native' 

418 executable = executable or 'ffmpeg' 

419 probefunc = fallback = None 

420 

421 if isinstance(method, str): 

422 probefunc = getattr(cls, '_probe_codec_' + method, None) 

423 if probefunc is None: 

424 raise AttributeError("Invalid probe method '%s'" % method) 

425 

426 if probefunc is cls._probe_codec_native: 

427 fallback = cls._probe_codec_fallback 

428 

429 elif callable(method): 

430 probefunc = method 

431 fallback = cls._probe_codec_fallback 

432 else: 

433 raise TypeError("Expected str or callable for parameter 'probe', " \ 

434 "not '{0.__class__.__name__}'" .format(method)) 

435 

436 codec = bitrate = None 

437 loop = asyncio.get_event_loop() 

438 try: 

439 codec, bitrate = await loop.run_in_executor(None, lambda: probefunc(source, executable)) 

440 except Exception: 

441 if not fallback: 

442 log.exception("Probe '%s' using '%s' failed", method, executable) 

443 return 

444 

445 log.exception("Probe '%s' using '%s' failed, trying fallback", method, executable) 

446 try: 

447 codec, bitrate = await loop.run_in_executor(None, lambda: fallback(source, executable)) 

448 except Exception: 

449 log.exception("Fallback probe using '%s' failed", executable) 

450 else: 

451 log.info("Fallback probe found codec=%s, bitrate=%s", codec, bitrate) 

452 else: 

453 log.info("Probe found codec=%s, bitrate=%s", codec, bitrate) 

454 finally: 

455 return codec, bitrate

457 @staticmethod 

458 def _probe_codec_native(source, executable='ffmpeg'): 

459 exe = executable[:2] + 'probe' if executable in ('ffmpeg', 'avconv') else executable 

460 args = [exe, '-v', 'quiet', '-print_format', 'json', '-show_streams', '-select_streams', 'a:0', source] 

461 output = subprocess.check_output(args, timeout=20) 

462 codec = bitrate = None 

463 

464 if output: 

465 data = json.loads(output) 

466 streamdata = data['streams'][0] 

467 

468 codec = streamdata.get('codec_name') 

469 bitrate = int(streamdata.get('bit_rate', 0)) 

470 bitrate = max(round(bitrate/1000, 0), 512) 

471 

472 return codec, bitrate

474 @staticmethod

546class AudioPlayer(threading.Thread): 

547 DELAY = OpusEncoder.FRAME_LENGTH / 1000.0 

548 

549 def __init__(self, source, client, *, after=None): 

550 threading.Thread.__init__(self) 

551 self.daemon = True 

552 self.source = source 

553 self.client = client 

554 self.after = after 

555 

556 self._end = threading.Event() 

557 self._resumed = threading.Event() 

558 self._resumed.set() # we are not paused 

559 self._current_error = None 

560 self._connected = client._connected 

561 self._lock = threading.Lock() 

562 

563 if after is not None and not callable(after): 

564 raise TypeError('Expected a callable for the "after" parameter.')

566 def _do_run(self): 

567 self.loops = 0 

568 self._start = time.perf_counter() 

569 

570 # getattr lookup speed ups 

571 play_audio = self.client.send_audio_packet 

572 self._speak(True) 

573 

574 while not self._end.is_set(): 

575 # are we paused? 

576 if not self._resumed.is_set(): 

577 # wait until we aren't 

578 self._resumed.wait() 

579 continue 

580 

581 # are we disconnected from voice? 

582 if not self._connected.is_set(): 

583 # wait until we are connected 

584 self._connected.wait() 

585 # reset our internal data 

586 self.loops = 0 

587 self._start = time.perf_counter() 

588 

589 self.loops += 1 

590 data = self.source.read() 

591 

592 if not data: 

593 self.stop() 

594 break 

595 

596 play_audio(data, encode=not self.source.is_opus()) 

597 next_time = self._start + self.DELAY * self.loops 

598 delay = max(0, self.DELAY + (next_time - time.perf_counter())) 

599 time.sleep(delay)

601 def run(self): 

602 try: 

603 self._do_run() 

604 except Exception as exc: 

605 self._current_error = exc 

606 self.stop()
